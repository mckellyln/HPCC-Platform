         ./../../system/security/shared
         ./../../esp/bindings/SOAP/xpp

         ./../../system/jlib
         ./../../esp/bindings

         ${CMAKE_BINARY_DIR}/oss

    install ( PROGRAMS testsocket.install DESTINATION etc/init.d/install COMPONENT Runtime )
